## Goal

Bring the server from a notification-in-memory prototype to a real multi-user backend: persistent data in Postgres via Prisma, JWT auth (HS256 access + refresh rotation with reuse detection), three feature modules (auth, devices, pairings), and a `/api/health` that actually probes the DB.

When this phase is done, two phones with the SyncWire app can register, log in, generate a pairing code from one, redeem it on the other, and the devices will be linked in the DB — ready for Phase 3 to start relaying notifications between them over MQTT.

**Non-goals for this phase:** no MQTT, no actual notification relay, no E2E encryption, no web client, no Android client, no rate-limiting beyond auth, no metrics, no graceful shutdown.

## Architecture

```
syncwire-server (Phase 1)
├── Postgres 18  ← already running in compose
│     ├── users            (id, email, password_hash, display_name, …)
│     ├── devices          (id, user_id, name, platform, push_token, …)
│     ├── refresh_tokens   (id, user_id, device_id, token_hash,
│     │                      expires_at, revoked_at, replaced_by_id, …)
│     ├── pairings         (id, source_user/device, viewer_user/device, status, …)
│     └── pairing_codes    (code, initiator_user/device, expires_at, used_at, …)
│
├── src/
│   ├── prisma/             ← NEW: PrismaService provider, schema mirror
│   ├── auth/               ← NEW: register, login, refresh, logout + JWT
│   ├── devices/            ← NEW: list, create, revoke
│   ├── pairings/           ← NEW: start, claim, list, revoke
│   ├── common/             ← NEW: @CurrentUser, @CurrentDevice, JwtAuthGuard
│   └── health/             ← UPDATED: add Postgres probe
│
├── test/                   ← UPDATED: e2e tests for every new endpoint
└── prisma/                 ← NEW: schema.prisma, migrations/
```

Two planes, one process:
- **REST control plane** (this phase): the new auth/devices/pairings modules plus the updated health probe.
- **MQTT data plane** (Phase 3): the bridge service that subscribes to `syncwire/relay/+` and republishes to per-viewer inboxes. Not touched here.

## Tech choices

| Concern | Choice | Why |
|---|---|---|
| ORM | Prisma 6.x (latest) | TypeScript-native, great DX, generated types, easy migrations; per `prisma.io` NestJS recipe |
| Migrations | `prisma migrate dev` (dev) / `prisma migrate deploy` (prod) | Forward-only additive in v1 |
| Hashing | `bcrypt` (cost 12) | Battle-tested, easy to install; argon2 would need extra Alpine deps |
| Auth | `@nestjs/jwt` + `@nestjs/passport` + `passport-jwt` | Standard NestJS pattern |
| JWT | HS256, 1h access TTL, 90d refresh TTL | Per `.env.example` (already in `src/config/env.ts`) |
| Refresh rotation | new token on every refresh; **reuse → revoke entire user session family** | Standard pattern (per OAuth 2.1 guidance) |
| Rate limiting | `@nestjs/throttler` | Idiomatic, already in plan §10 |
| Testing (unit) | Jest + `@nestjs/testing` | Already in stack |
| Testing (e2e) | supertest + real Postgres (compose service) | Already in stack; tests run against the actual DB |
| UUIDs | `crypto.randomUUID()` for ids, Postgres `gen_random_uuid()` (pgcrypto) for defaults | Idempotent, no `cuid2` dep needed |
| Validation | class-validator (existing) | Consistent with `CreateNotificationDto` |
| Schema migration runner | One-shot `app-migrate` container in compose that runs `prisma migrate deploy` before `app` starts | Avoids the "app starts before DB is ready" race |

```prisma
// SyncWire — Phase 1 schema
// Tables: users, devices, refresh_tokens, pairings, pairing_codes

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum DevicePlatform {
  android
  web
}

enum PairingStatus {
  active
  revoked
}

model User {
  id            String   @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  email         String   @unique
  passwordHash  String   @map("password_hash")
  displayName   String   @map("display_name")
  createdAt     DateTime @default(now()) @map("created_at")
  updatedAt     DateTime @updatedAt        @map("updated_at")

  devices       Device[]
  refreshTokens RefreshToken[]

  @@map("users")
}

model Device {
  id          String         @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  userId      String         @map("user_id") @db.Uuid
  name        String
  platform    DevicePlatform
  pushToken   String?        @map("push_token")
  lastSeenAt  DateTime?      @map("last_seen_at")
  createdAt   DateTime       @default(now()) @map("created_at")
  revokedAt   DateTime?      @map("revoked_at")

  user                User              @relation(fields: [userId], references: [id], onDelete: Cascade)
  refreshTokens       RefreshToken[]
  pairingsAsSource    Pairing[]         @relation("PairingSource")
  pairingsAsViewer    Pairing[]         @relation("PairingViewer")
  pairingCodesInitiated PairingCode[]   @relation("PairingCodeInitiator")

  @@index([userId])
  @@map("devices")
}

model RefreshToken {
  id            String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  userId        String    @map("user_id") @db.Uuid
  deviceId      String    @map("device_id") @db.Uuid
  // SHA-256 hex of the opaque token string (64 chars). NOT the JWT.
  tokenHash     String    @unique @map("token_hash")
  expiresAt     DateTime  @map("expires_at")
  // Non-null when this token was rotated (replaced by a newer one).
  replacedById  String?   @unique @map("replaced_by_id") @db.Uuid
  // Set on /auth/logout or on reuse-detection.
  revokedAt     DateTime? @map("revoked_at")
  createdAt     DateTime  @default(now()) @map("created_at")

  user   User   @relation(fields: [userId], references: [id], onDelete: Cascade)
  device Device @relation(fields: [deviceId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([deviceId])
  @@index([expiresAt])
  @@map("refresh_tokens")
}

model Pairing {
  id              String        @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  sourceUserId    String        @map("source_user_id") @db.Uuid
  sourceDeviceId  String        @map("source_device_id") @db.Uuid
  viewerUserId    String        @map("viewer_user_id") @db.Uuid
  viewerDeviceId  String        @map("viewer_device_id") @db.Uuid
  status          PairingStatus @default(active)
  createdAt       DateTime      @default(now()) @map("created_at")
  revokedAt       DateTime?     @map("revoked_at")

  sourceUser   User   @relation("PairingSource",  fields: [sourceUserId],   references: [id], onDelete: Cascade)
  sourceDevice Device @relation("PairingSource",  fields: [sourceDeviceId], references: [id], onDelete: Cascade)
  viewerUser   User   @relation("PairingViewer",  fields: [viewerUserId],   references: [id], onDelete: Cascade)
  viewerDevice Device @relation("PairingViewer",  fields: [viewerDeviceId], references: [id], onDelete: Cascade)

  // Prevent duplicate active pairings between the same two devices.
  // (Revoked ones can be re-created.)
  @@unique([sourceDeviceId, viewerDeviceId])
  @@index([sourceUserId])
  @@index([viewerUserId])
  @@index([status])
  @@map("pairings")
}

model PairingCode {
  // 6-digit numeric code, stored as a string to preserve leading zeros.
  code              String    @id @map("code")
  initiatorUserId   String    @map("initiator_user_id") @db.Uuid
  initiatorDeviceId String    @map("initiator_device_id") @db.Uuid
  expiresAt         DateTime  @map("expires_at")
  usedAt            DateTime? @map("used_at")
  createdAt         DateTime  @default(now()) @map("created_at")

  initiatorUser   User   @relation("PairingCodeInitiatorUser",   fields: [initiatorUserId],   references: [id], onDelete: Cascade)
  initiatorDevice Device @relation("PairingCodeInitiator",       fields: [initiatorDeviceId], references: [id], onDelete: Cascade)

  @@index([initiatorUserId])
  @@index([expiresAt])
  @@map("pairing_codes")
}
```

```ts
import { Controller, Get } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';

type CheckResult = { status: 'ok' | 'down'; detail?: string };

@Controller('health')
export class HealthController {
  constructor(private readonly prisma: PrismaService) {}

  @Get()
  async check(): Promise<{
    status: 'ok' | 'down';
    uptimeSeconds: number;
    timestamp: string;
    checks: Record<string, CheckResult>;
  }> {
    const checks: Record<string, CheckResult> = {
      database: await this.probeDb(),
      mqtt: { status: 'ok', detail: 'not_probed_yet' }, // Phase 2
    };

    const allOk = Object.values(checks).every((c) => c.status === 'ok');
    return {
      status: allOk ? 'ok' : 'down',
      uptimeSeconds: Math.round(process.uptime()),
      timestamp: new Date().toISOString(),
      checks,
    };
  }

  private async probeDb(): Promise<CheckResult> {
    try {
      await this.prisma.$queryRaw`SELECT 1`;
      return { status: 'ok' };
    } catch (e) {
      return { status: 'down', detail: (e as Error).message };
    }
  }
}
```

```ts
import { Injectable } from '@nestjs/common';
import * as bcrypt from 'bcrypt';

@Injectable()
export class HashService {
  private readonly COST = 12;

  hash(plain: string): Promise<string> {
    return bcrypt.hash(plain, this.COST);
  }

  verify(plain: string, hash: string): Promise<boolean> {
    return bcrypt.compare(plain, hash);
  }
}
```

**Implementation:**
```ts
import { Injectable, Inject } from '@nestjs/common';
import { JwtService as NestJwtService } from '@nestjs/jwt';
import { JwtModuleOptions } from './jwt.options';
import { JWT_OPTIONS } from './jwt.constants';

export interface AccessTokenPayload {
  sub: string;       // user id
  devices: string[]; // device ids owned by this user
  iat?: number;
  exp?: number;
}

@Injectable()
export class JwtService {
  constructor(
    private readonly nestJwt: NestJwtService,
    @Inject(JWT_OPTIONS) private readonly opts: JwtModuleOptions,
  ) {}

  signAccess(payload: Omit<AccessTokenPayload, 'iat' | 'exp'>): Promise<string> {
    return this.nestJwt.signAsync(payload, {
      secret: this.opts.secret,
      expiresIn: this.opts.accessTtlSeconds,
    });
  }

  verifyAccess(token: string): Promise<AccessTokenPayload> {
    return this.nestJwt.verifyAsync(token, { secret: this.opts.secret });
  }
}
```

(Refresh tokens are opaque random strings, not JWTs — they live in the `refresh_tokens` table. See M2-T7.)

**Options file** (`jwt.options.ts`):
```ts

  accessTtlSeconds: number;
  refreshTtlSeconds: number;
```

```ts
import { Injectable, ConflictException, UnauthorizedException, Inject } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { HashService } from '../common/hash/hash.service';
import { JwtService } from './jwt.service';
import { JWT_OPTIONS } from './jwt.constants';
import { JwtModuleOptions } from './jwt.options';
import { RegisterDto } from './dto/register.dto';
import { LoginDto } from './dto/login.dto';
import { DevicePlatform, User } from '@prisma/client';
import { randomBytes, randomUUID } from 'crypto';

export interface AuthResult {
  accessToken: string;
  refreshToken: string;
  refreshTokenId: string;
  user: { id: string; email: string; displayName: string };
}

export interface DeviceInfo {
  name: string;
  platform: DevicePlatform;
}

const REFRESH_BYTES = 48; // 384 bits

@Injectable()
export class AuthService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly hash: HashService,
    private readonly jwt: JwtService,
    @Inject(JWT_OPTIONS) private readonly opts: JwtModuleOptions,
  ) {}

  async register(dto: RegisterDto, device: DeviceInfo): Promise<AuthResult> {
    const existing = await this.prisma.user.findUnique({ where: { email: dto.email } });
    if (existing) throw new ConflictException('email already registered');

    const passwordHash = await this.hash.hash(dto.password);
    const user = await this.prisma.user.create({
      data: {
        email: dto.email,
        passwordHash,
        displayName: dto.displayName,
      },
    });
    return this.issueTokens(user, device);
  }

  async login(dto: LoginDto, device: DeviceInfo): Promise<AuthResult> {
    const user = await this.prisma.user.findUnique({ where: { email: dto.email } });
    if (!user) throw new UnauthorizedException('invalid credentials');

    const ok = await this.hash.verify(dto.password, user.passwordHash);
    if (!ok) throw new UnauthorizedException('invalid credentials');

    return this.issueTokens(user, device);
  }

  /** Shared path for register + login. */
  private async issueTokens(user: User, device: DeviceInfo): Promise<AuthResult> {
    const createdDevice = await this.prisma.device.create({
      data: {
        userId: user.id,
        name: device.name,
        platform: device.platform,
      },
    });

    const accessToken = await this.jwt.signAccess({
      sub: user.id,
      devices: [createdDevice.id],
    });

    const refreshToken = randomBytes(REFRESH_BYTES).toString('base64url');
    const refreshTokenHash = await this.hash.hash(refreshToken);
    const expiresAt = new Date(Date.now() + this.opts.refreshTtlSeconds * 1000);
    const refreshRow = await this.prisma.refreshToken.create({
      data: {
        userId: user.id,
        deviceId: createdDevice.id,
        tokenHash: refreshTokenHash,
        expiresAt,
      },
    });

    return {
      accessToken,
      refreshToken,
      refreshTokenId: refreshRow.id,
      user: { id: user.id, email: user.email, displayName: user.displayName },
    };
  }
}
```

```ts
import { JwtService } from '../../auth/jwt.service';

@Injectable()
export class JwtAuthGuard implements CanActivate {
  constructor(private readonly jwt: JwtService) {}

  async canActivate(ctx: ExecutionContext): Promise<boolean> {
    const req = ctx.switchToHttp().getRequest();
    const auth = req.headers['authorization'] as string | undefined;
    if (!auth || !auth.startsWith('Bearer ')) {
      throw new UnauthorizedException('missing bearer token');
    }
    const token = auth.slice('Bearer '.length);
    try {
      req.user = await this.jwt.verifyAccess(token);
    } catch {
      throw new UnauthorizedException('invalid or expired token');
    }
    return true;
  }
}
```

```ts
import { CreateDeviceDto } from './dto/create-device.dto';

@Injectable()
export class DevicesService {
  constructor(private readonly prisma: PrismaService) {}

  list(userId: string) {
    return this.prisma.device.findMany({
      where: { userId, revokedAt: null },
      orderBy: { createdAt: 'desc' },

  create(userId: string, dto: CreateDeviceDto) {

  async revoke(userId: string, deviceId: string) {
    const device = await this.prisma.device.findFirst({
      where: { id: deviceId, userId, revokedAt: null },
    });
    if (!device) throw new NotFoundException('device not found');
    await this.prisma.device.update({
      where: { id: deviceId },
      data: { revokedAt: new Date() },
    });
    // Cascade-revoke this device's refresh tokens
    await this.prisma.refreshToken.updateMany({
      where: { deviceId, revokedAt: null },
      data: { revokedAt: new Date() },
    });
  }
```

### M4-T2: PairingsService — code generation + atomic claim (the tricky part)

**Files:**
- Create: `src/pairings/pairings.service.ts`
- Test: `src/pairings/pairings.service.spec.ts`

**Unit tests** to cover:
- `start(userId, sourceDeviceId)` creates a 6-digit code, stores its SHA-256 hash, returns the plaintext
- `claim(code, viewer)` finds the row, checks the hash, checks expiry, checks not already used, creates the Pairing row, marks `usedAt`
- `claim` with wrong code → NotFoundException
- `claim` with expired code → GoneException
- `claim` with already-used code → ConflictException
- `claim` is atomic: two concurrent calls with the same code — exactly one succeeds, the other gets ConflictException
- `revoke(userId, pairingId)` marks the pairing revoked (either side can revoke)
- `list(userId)` returns pairings where the user is source OR viewer, active only

**Implementation** (key pieces):

```ts
import { Injectable, NotFoundException, ConflictException, GoneException, BadRequestException, Inject } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { JwtModuleOptions } from '../auth/jwt.options';
import { JWT_OPTIONS } from '../auth/jwt.constants';
import { randomInt } from 'crypto';
import { createHash } from 'crypto';
import { StartPairingDto } from './dto/start-pairing.dto';
import { ClaimPairingDto } from './dto/claim-pairing.dto';
import { DevicePlatform } from '@prisma/client';

const CODE_TTL_MS = 10 * 60 * 1000; // 10 minutes

@Injectable()
export class PairingsService {
  constructor(
    private readonly prisma: PrismaService,
    @Inject(JWT_OPTIONS) private readonly opts: JwtModuleOptions, // for future use
  ) {}

  /** Start a pairing: generate a 6-digit code, store its SHA-256 hash, return the plaintext. */
  async start(initiatorUserId: string, dto: StartPairingDto): Promise<{ code: string; expiresAt: Date; pairingId: string }> {
    // Confirm the source device belongs to the initiator
    const device = await this.prisma.device.findFirst({
      where: { id: dto.sourceDeviceId, userId: initiatorUserId, revokedAt: null },
    });
    if (!device) throw new NotFoundException('source device not found or not yours');

    // Generate 6-digit code, cryptographically random
    const code = randomInt(0, 1_000_000).toString().padStart(6, '0');
    const codeHash = createHash('sha256').update(code).digest('hex');
    const expiresAt = new Date(Date.now() + CODE_TTL_MS);

    // Upsert pattern: delete any existing code for this user first to avoid
    // two active codes per user (which is confusing).
    await this.prisma.pairingCode.deleteMany({ where: { initiatorUserId } });
    const row = await this.prisma.pairingCode.create({
      data: {
        code: codeHash,        // store HASH not plaintext
        initiatorUserId,
        initiatorDeviceId: dto.sourceDeviceId,
        expiresAt,
      },
    });

    return { code, expiresAt, pairingId: row.code };
  }

  /** Claim a pairing: atomic transaction. Returns the created pairing. */
  async claim(viewerUserId: string, dto: ClaimPairingDto): Promise<{ pairingId: string }> {
    const codeHash = createHash('sha256').update(dto.code).digest('hex');

    // Atomic transaction to prevent two clients from claiming the same code.
    const result = await this.prisma.$transaction(async (tx) => {
      const code = await tx.pairingCode.findUnique({ where: { code: codeHash } });
      if (!code) throw new NotFoundException('invalid code');
      if (code.usedAt) throw new ConflictException('code already used');
      if (code.expiresAt.getTime() < Date.now()) throw new GoneException('code expired');
      if (code.initiatorUserId === viewerUserId) {
        throw new BadRequestException('cannot pair with yourself');
      }

      // Create the viewer's device (Phase 1 simplification: no separate device-create)
      const viewerDevice = await tx.device.create({
        data: {
          userId: viewerUserId,
          name: dto.viewerDeviceName,
          platform: dto.viewerDevicePlatform,
        },
      });

      // Create the pairing (unique constraint on (sourceDeviceId, viewerDeviceId)
      // prevents duplicates — caught below as P2002)
      let pairing;
      try {
        pairing = await tx.pairing.create({
          data: {
            sourceUserId: code.initiatorUserId,
            sourceDeviceId: code.initiatorDeviceId,
            viewerUserId,
            viewerDeviceId: viewerDevice.id,
            status: 'active',
          },
        });
      } catch (e: any) {
        if (e?.code === 'P2002') {
          throw new ConflictException('devices are already paired');
        }
        throw e;
      }

      // Mark code used
      await tx.pairingCode.update({
        where: { code: codeHash },
        data: { usedAt: new Date() },
      });

      return pairing;
    });

    return { pairingId: result.id };
  }

  list(userId: string) {
    return this.prisma.pairing.findMany({
      where: {
        OR: [
          { sourceUserId: userId, status: 'active' },
          { viewerUserId: userId, status: 'active' },
        ],
      },
      orderBy: { createdAt: 'desc' },
    });
  }

  async revoke(userId: string, pairingId: string): Promise<void> {
    const pairing = await this.prisma.pairing.findFirst({
      where: {
        id: pairingId,
        status: 'active',
        OR: [{ sourceUserId: userId }, { viewerUserId: userId }],
      },
    });
    if (!pairing) throw new NotFoundException('pairing not found');
    await this.prisma.pairing.update({
      where: { id: pairingId },
      data: { status: 'revoked', revokedAt: new Date() },
    });
  }
}
```
